Hangul.js 된소리 예제

  • 자모음 분리
  • 된소리
  • 난독화

이 예제는 입력으로 주어진 문장에 있는 예사소리와, 거센소리를 모두 된소리로 바꾼 문장을 보여줍니다.

입력 출력
ㄱ, ㅋㄲ
ㄷ, ㅌㄸ
ㅂ, ㅍㅃ
ㅅㅆ
ㅈ, ㅊㅉ
  • 가나다라마바사아자차카타파하 → 까나따라마빠싸아짜짜까따빠하
  • 도레미파솔라시도 → 또레미빠쏠라씨또

코드

function stronger(x){
  if(x == 'ㄱ' || x == 'ㅋ') return 'ㄲ';
  if(x == 'ㄷ' || x == 'ㅌ') return 'ㄸ';
  if(x == 'ㅂ' || x == 'ㅍ') return 'ㅃ';
  if(x == 'ㅅ') return 'ㅆ';
  if(x == 'ㅈ' || x == 'ㅊ') return 'ㅉ';
  return x;
}

$('#output').innerHTML = Hangul.assemble(Hangul.disassemble(input).map(stronger));
